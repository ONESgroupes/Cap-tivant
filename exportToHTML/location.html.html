<script>
    function toggleMenu() {
        const menu = document.getElementById("menu-overlay");
        if (menu.style.display === "flex") {
            menu.style.display = "none";
        } else {
            menu.style.display = "flex";
        }
    }

    let typeSelectionne = "";

    function selectType(button) {
        const boutonClique = button.textContent.trim().toLowerCase(); // "moteur" ou "à voile"
        const lienRecherche = document.getElementById("lien-recherche");

        // Si on reclique sur le même bouton déjà actif, on désactive tout
        if (button.classList.contains("active")) {
            button.classList.remove("active");
            typeSelectionne = "";
            lienRecherche.href = "offre.html";
        } else {
            // On désactive les autres boutons
            document.querySelectorAll('.bouton-type').forEach(btn => btn.classList.remove('active'));
            // On active celui qu'on vient de cliquer
            button.classList.add("active");
            typeSelectionne = boutonClique;

            if (typeSelectionne === "moteur") {
                lienRecherche.href = "offre.html?type=moteur";
            } else if (typeSelectionne === "à voile") {
                lienRecherche.href = "offre.html?type=voile";
            } else {
                lienRecherche.href = "offre.html";
            }
        }
    }
</script>
