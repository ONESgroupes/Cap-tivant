<script>
    function toggleMenu() {
        const menu = document.getElementById("menu-overlay");
        if (menu.style.display === "flex") {
            menu.style.display = "none";
        } else {
            menu.style.display = "flex";
        }
    }
</script>

<script>
    const carrouselIndexes = new Map(); // Pour suivre chaque carrousel

    function changeSlide(button, direction) {
        const carrousel = button.closest('.carrousel');
        const slides = carrousel.querySelectorAll('.slide');

        if (!carrouselIndexes.has(carrousel)) {
            carrouselIndexes.set(carrousel, 0);
        }

        let currentIndex = carrouselIndexes.get(carrousel);
        slides[currentIndex].classList.remove('active');

        currentIndex = (currentIndex + direction + slides.length) % slides.length;

        slides[currentIndex].classList.add('active');
        carrouselIndexes.set(carrousel, currentIndex);
    }
</script>


<script>
    window.addEventListener('DOMContentLoaded', () => {
        const params = new URLSearchParams(window.location.search);
        const type = params.get('type');

        if (type === "moteur" || type === "voile") {
            document.querySelectorAll('.carrousel').forEach(carrousel => {
                if (carrousel.classList.contains(type)) {
                    carrousel.style.display = "flex";
                } else {
                    carrousel.style.display = "none";
                }
            });
        } else {
            // Si aucun type n'est précisé, tout afficher
            document.querySelectorAll('.carrousel').forEach(carrousel => {
                carrousel.style.display = "flex";
            });
        }
    });
</script>
